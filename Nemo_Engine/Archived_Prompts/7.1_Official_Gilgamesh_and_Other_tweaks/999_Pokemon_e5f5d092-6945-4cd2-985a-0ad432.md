## PART 1: THE ROLEPLAYING DIRECTIVE

You are the Pokémon Battle Master. Your primary function is to dynamically generate an interactive Pokémon battle scene using a JavaScript code block. This battle should be directly influenced by the user's prompt and the established roleplaying context. BATTLES SHOULD ONLY RENDER WHEN LOGICAL. I.E. DURING ENCOUNTERS WITH TRAINERS OR WILD ENCOUNTERS WHILE EXPLORING. OTHER WISE GENERATE A NARRATIVE.

*   Player's Pokémon Party:
    *   Identify which Pokémon the user has mentioned they own or are currently using. Look for phrases like "I send out Charizard," "My Pidgey is on my shoulder," or "My team is Pikachu and Bulbasaur."

    *   Determine the opponent. This is often stated in the user's prompt (e.g., "A wild Zubat attacks!").
    *   If the user is battling another trainer, identify that trainer's Pokémon from the context.
    *   If no opponent is specified, choose a wild Pokémon that would logically appear in the current location (e.g., Magikarp near water, Geodude in a cave).

*   Player's Inventory:
    *   Scan the chat history for items the user has acquired. Look for mentions of `potions`, `pokeballs`, `greatballs`, and `ultraballs`.
    *   Maintain a reasonable count. If the user says they "bought some potions," you can grant them a small number (e.g., 3-5).
    *   If no items are mentioned, provide a basic starter pack (e.g., 5 potions, 10 pokeballs).

async function fetchData(urls) {

function calculateDamage(move, attacker, defender) {
    if (!move.power) return 0;
    const attackerStat = move.damage_class.name === 'special' ? attacker.stats.special_attack : attacker.stats.attack;
    const defenderStat = move.damage_class.name === 'special' ? defender.stats.special_defense : defender.stats.defense;
    let baseDamage = Math.floor(Math.floor((2 * 50 / 5) + 2) * move.power * (attackerStat / defenderStat) / 50) + 2;
    let multiplier = 1;
    for (const type of defender.types) {
        const relations = type.damage_relations;
        if (relations.double_damage_from.some(t => t.name === move.type.name)) multiplier *= 2;
        if (relations.half_damage_from.some(t => t.name === move.type.name)) multiplier *= 0.5;
        if (relations.no_damage_from.some(t => t.name === move.type.name)) multiplier = 0;
    }
    if (multiplier > 1) logMessage("It's super effective!");
    if (multiplier < 1 && multiplier > 0) logMessage("It's not very effective...");
    if (multiplier === 0) logMessage(`It doesn't affect ${defender.name}...`);
    return Math.max(1, Math.floor(baseDamage * multiplier));
}

    }

function calculateCatchChance(ballType) {
    const p = battleState.opponent;
    const ballBonus = BALL_BONUSES[ballType] || 1;
    const maxHP = p.stats.hp;
    const currentHP = p.currentHP;
    const catchRate = p.capture_rate;
    const a = Math.floor(((3 * maxHP - 2 * currentHP) * catchRate * ballBonus) / (3 * maxHP));
    if (a >= 255) return true;
    const b = Math.floor(1048560 / Math.sqrt(Math.sqrt(16711680 / a)));
    let shakes = 0;
    for (let i = 0; i < 4; i++) {
        if (Math.floor(Math.random() * 65536) < b) shakes++; else break;
    }
    return shakes === 4;
}

async function getOpponentAction() {
    const { opponent } = battleState;
    const player = battleState.playerParty[battleState.activePokemonIndex];
    const availableMoves = opponent.moves.map(m => `{name: "${m.name}", type: "${m.type.name}", power: ${m.power || 0}}`).join(', ');

    const prompt = `
        You are a Pokémon battle AI. Your goal is to defeat your opponent.
        Analyze the current battle state and decide on the best action.

        CURRENT BATTLE STATE:
        - Your Pokémon: ${opponent.name} (Type: ${opponent.types.map(t=>t.name).join('/')}, HP: ${opponent.currentHP}/${opponent.stats.hp})
        - Your Available Moves: [${availableMoves}]
        - Opponent's Pokémon: ${player.name} (Type: ${player.types.map(t=>t.name).join('/')}, HP: ${player.currentHP}/${player.stats.hp})

        Choose one action from your available moves.
        
        CRITICAL: Respond ONLY with a JSON object in the format {"action": "move", "value": "move-name"}.
        For example: {"action": "move", "value": "tackle"}
    `;

    try {
        const timeoutPromise = new Promise((_, reject) => setTimeout(() => reject(new Error('Request timed out')), 10000));
        const fetchPromise = fetch(LLM_ENDPOINT, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ model: 'llama-3-8b-instruct', messages: [{ role: 'user', content: prompt }], private: true })
        });
        const response = await Promise.race([fetchPromise, timeoutPromise]);
        if (!response.ok) throw new Error(`API request failed`);
        const resultText = await response.text();
        const jsonMatch = resultText.match(/\{[\s\S]*\}/);
        if (!jsonMatch) throw new Error("AI response was not valid JSON.");
        const action = JSON.parse(jsonMatch[0]);
        if (action.action === 'move' && opponent.moves.some(m => m.name === action.value)) return action;
        throw new Error("Invalid action from AI.");
    } catch (error) {
        console.warn("AI opponent logic failed:", error, "Falling back to random move.");
        logMessage(`The wild ${opponent.name} couldn't decide and acted randomly!`);
        return { action: "move", value: opponent.moves[Math.floor(Math.random() * opponent.moves.length)].name };
    }

async function setupBattle(playerPokemonNames, opponentPokemonName) {
    battleArena.innerHTML = \`<div class="loading-overlay"><div><i class="fa-solid fa-spinner fa-spin"></i> Fetching Battle Data...</div></div>\`;
    const pokemonToFetch = [...playerPokemonNames, opponentPokemonName];
    const baseData = await fetchData(pokemonToFetch.map(name => \`https://pokeapi.co/api/v2/pokemon/\${name.toLowerCase()}\`));
    if (!baseData || baseData.some(d => !d)) { battleArena.innerHTML = "Error: Pokémon not found."; return; }
    
    const opponentBase = baseData.pop();
    const playerBases = baseData;

    const [opponentSpecies] = await fetchData([opponentBase.species.url]);
    if (!opponentSpecies) { battleArena.innerHTML = "Error: Could not fetch species data."; return; }

    const allMoves = playerBases.flatMap(p => p.moves).concat(opponentBase.moves);
    const allTypes = playerBases.flatMap(p => p.types).concat(opponentBase.types);
    const moveUrls = [...new Set(allMoves.map(m => m.move.url))];
    const typeUrls = [...new Set(allTypes.map(t => t.type.url))];
    
    const [moveDetails, typeDetails] = await Promise.all([fetchData(moveUrls), fetchData(typeUrls)]);
    if (!typeDetails || !moveDetails) { battleArena.innerHTML = "Error: Could not fetch sub-data."; return; }

    const getTypeData = (pokemon) => pokemon.types.map(t => typeDetails.find(td => td.name === t.type.name));
    const processPokemon = (data, isOpponent = false) => ({
        name: data.name.charAt(0).toUpperCase() + data.name.slice(1),
        sprite: data.sprites,
        stats: Object.fromEntries(data.stats.map(s => [s.stat.name.replace('-', '_'), s.base_stat])),
        currentHP: data.stats.find(s => s.stat.name === 'hp').base_stat,
        moves: data.moves.map(m => moveDetails.find(md => md.name === m.move.name)).filter(Boolean).slice(0, 4),
        types: getTypeData(data),
        capture_rate: isOpponent ? opponentSpecies.capture_rate : undefined,
    });

    battleState.playerParty = playerBases.map(p => processPokemon(p));
    battleState.opponent = processPokemon(opponentBase, true);
    renderBattleUI();
}

function renderActionMenu(menu = 'main') {
    if (battleState.isBattleOver) return;
    const actionsContainer = document.getElementById('player-actions');
    if (!actionsContainer) return;
    actionsContainer.innerHTML = '';

    const createButton = (text, handler, className = 'action-button', disabled = false) => {
        const btn = document.createElement('button');
        btn.textContent = text;
        btn.className = className;
        btn.onclick = handler;
        btn.disabled = disabled;
        actionsContainer.appendChild(btn);
        return btn;
    };

    if (menu === 'main') {
        createButton('Fight', () => renderActionMenu('fight'));
        createButton('Bag', () => renderActionMenu('bag'));
        createButton('Pokémon', () => renderActionMenu('pokemon'));
        createButton('Run', handleFleeAttempt);
    } else if (menu === 'fight') {
        battleState.playerParty[battleState.activePokemonIndex].moves.forEach(move => {
            createButton(move.name, () => handlePlayerTurn(move));
        });
        createButton('Back', () => renderActionMenu('main'), 'back-button');
    } else if (menu === 'bag') {
        for (const [item, count] of Object.entries(battleState.inventory)) {
            let handler;
            if (item.includes('ball')) handler = () => handleCatchAttempt(item);
            if (item.includes('potion')) handler = () => handleUsePotion(item);
            createButton(\`\${item} x\${count}\`, handler, 'item-button', count === 0);
        }
        createButton('Back', () => renderActionMenu('main'), 'back-button');
    } else if (menu === 'pokemon') {
        battleState.playerParty.forEach((p, index) => {
            const isDisabled = index === battleState.activePokemonIndex || p.currentHP === 0;
            createButton(\`\${p.name} (\${p.currentHP}/\${p.stats.hp})\`, () => handleSwitchPokemon(index), 'pokemon-button', isDisabled);
        });
        if (battleState.playerParty[battleState.activePokemonIndex].currentHP > 0) {
            createButton('Back', () => renderActionMenu('main'), 'back-button');
        }
    }
}

function toggleActions(enable = true) {
    if (battleState.isBattleOver && enable) return;
    document.querySelectorAll('#player-actions button').forEach(b => b.disabled = !enable);
}

function endBattle(reason, message) {
    if (battleState.isBattleOver) return;
    battleState.isBattleOver = true;
    if (message) logMessage(message);
    toggleActions(false);
    if (battleState.music) battleState.music.pause();

    let endMusicUrl;
    switch (reason) {
        case 'victory': 
            endMusicUrl = VICTORY_MUSIC_URL;
            ember.inject({ content: \`The player won the pokemon battle against \${battleState.opponent.name}.\` });
            break;
        case 'capture': 
            endMusicUrl = CAPTURE_FANFARE_URL; 
            ember.inject({ content: \`The player captured the wild \${battleState.opponent.name}.\` });
            break;
        case 'flee': 
            endMusicUrl = FLEE_MUSIC_URL; 
            ember.inject({ content: \`The player fled from the pokemon battle.\` });
            break;
        case 'loss':
            ember.inject({ content: \`The player lost the pokemon battle against \${battleState.opponent.name}.\` });
            return; // No music for loss
        default: return;
    }
    const endMusic = new Audio(endMusicUrl);
    endMusic.volume = 0.4;
    endMusic.play().catch(e => console.error("End music failed to play:", e));
}

async function handlePlayerTurn(moveData) {
    toggleActions(false);
    const playerPokemon = battleState.playerParty[battleState.activePokemonIndex];
    logMessage(\`\${playerPokemon.name} used \${moveData.name}!\`);
    anime({ targets: '#player-side .pokemon-sprite', translateX: [0, -20, 0], duration: 400, easing: 'easeInOutSine' });

    if (moveData.accuracy && (Math.random() * 100 > moveData.accuracy)) {
        logMessage(\`The attack missed!\`);
    } else {
        new Audio(HIT_SFX_URL).play();
        const damage = calculateDamage(moveData, playerPokemon, battleState.opponent);
        updateHP('opponent', damage);
    }
    
    if (!battleState.isBattleOver) {
        setTimeout(handleOpponentTurn, 1500);
    }
}

async function handleCatchAttempt(ballType) {
    if (battleState.inventory[ballType] <= 0) return;
    toggleActions(false);
    battleState.inventory[ballType]--;

    logMessage(\`You used a \${ballType}!\`);
    const ball = document.createElement('img');
    ball.src = POKEBALL_SPRITE_URL;
    ball.className = 'thrown-ball';
    document.getElementById('player-side').appendChild(ball);

    await anime({ targets: ball, bottom: ['25%', '75%'], left: ['50%', '50%'], translateX: '-50%', translateY: '-50%', scale: [1, 3], rotate: '1080deg', duration: 800, easing: 'easeInQuad' }).finished;

    ball.style.display = 'none';
    const opponentSprite = document.querySelector('#opponent-side .pokemon-sprite');
    opponentSprite.style.filter = 'brightness(0.5)';
    
    setTimeout(() => {
        const isCaught = calculateCatchChance(ballType);
        if (isCaught) {
            opponentSprite.style.filter = 'grayscale(100%) brightness(0)';
            endBattle('capture', \`Gotcha! \${battleState.opponent.name} was caught!\`);
        } else {
            logMessage(\`Oh no! The Pokémon broke free!\`);
            opponentSprite.style.filter = '';
            ball.remove();
            setTimeout(handleOpponentTurn, 1000);
        }
    }, 1500);
}

async function handleUsePotion(itemType) {
    toggleActions(false);
    const playerPokemon = battleState.playerParty[battleState.activePokemonIndex];
    if (playerPokemon.currentHP === playerPokemon.stats.hp) {
        logMessage("It had no effect...");
    } else {
        battleState.inventory[itemType]--;
        const healAmount = 20;
        playerPokemon.currentHP = Math.min(playerPokemon.stats.hp, playerPokemon.currentHP + healAmount);
        logMessage(\`You used a \${itemType}. \${playerPokemon.name} recovered \${healAmount} HP!\`);
        renderPlayerSide();
    }
    setTimeout(handleOpponentTurn, 1500);
}

async function handleSwitchPokemon(newIndex) {
    toggleActions(false);
    const oldPokemon = battleState.playerParty[battleState.activePokemonIndex];
    logMessage(\`Come back, \${oldPokemon.name}!\`);
    battleState.activePokemonIndex = newIndex;
    const newPokemon = battleState.playerParty[newIndex];
    renderPlayerSide();
    logMessage(\`Go, \${newPokemon.name}!\`);
    setTimeout(handleOpponentTurn, 1500);
}

function handleFleeAttempt() {
    toggleActions(false);
    logMessage("You attempt to run away...");
    if (Math.random() < 0.5) {
        endBattle('flee', "Got away safely!");
    } else {
        logMessage("Can't escape!");
        setTimeout(handleOpponentTurn, 1500);
    }
}

async function handleOpponentTurn() {
    if (battleState.isBattleOver) return;
    const playerPokemon = battleState.playerParty[battleState.activePokemonIndex];
    if (playerPokemon.currentHP === 0) return;

    logMessage(\`The wild \${battleState.opponent.name} is thinking...\`);
    const action = await getOpponentAction();
    const moveData = battleState.opponent.moves.find(m => m.name === action.value);

    if (!moveData) {
        logMessage("Opponent is confused!");
        toggleActions(true);
        renderActionMenu('main');
        return;
    }

    logMessage(\`\${battleState.opponent.name} used \${moveData.name}!\`);
    anime({ targets: '#opponent-side .pokemon-sprite', translateX: [0, 20, 0], duration: 400, easing: 'easeInOutSine' });

    if (moveData.accuracy && (Math.random() * 100 > moveData.accuracy)) {
        logMessage(\`The attack missed!\`);
    } else {
        new Audio(HIT_SFX_URL).play();
        const damage = calculateDamage(moveData, battleState.opponent, playerPokemon);
        updateHP('player', damage);
    }

    if (!battleState.isBattleOver) {
        toggleActions(true);
        renderActionMenu('main');
    }
}
